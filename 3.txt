<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true? - Потому что оператор == приводит операнды к одному типу, а потом сравнивает их значения
    var_dump((int)'012345');     // Почему 12345? - Потому что int приводит к целому числу
    var_dump((float)123.0 === (int)123.0); // Почему false? - Потому что оператор === сравнивает типы данных, а float и int разные типы данных
    var_dump((int)0 === (int)'hello, world'); // Почему true? - Все, что не является чилом int приводит к 0, 0 === 0 true
?>
